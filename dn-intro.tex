\section{课程简介}
%% 课程 
%% 目标 内容

\begin{frame}
\frametitle{课程简介}

\CJKindent 了解 ~.NET 框架的功能，学习 ~C\# 语言，学习 ~.NET 框架类库， 学习
~.NET 框架的开发。

\begin{itemize}
\item .NET 平台和 C\# 基础
\begin{itemize}
\item .NET 框架，公共语言运行时 
\item C\# 语言，类型基础，面向对象开发
\end{itemize}
\item 框架类库基础
  \begin{itemize}
  \item 文件 ~I/O，数据流
  \item 多线程，网络编程
  \end{itemize}
\item 图形界面开发
\item ASP.NET 
\item ADO.NET
\end{itemize}
\end{frame}
